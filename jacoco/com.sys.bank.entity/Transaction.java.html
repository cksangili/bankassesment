package com.sys.bank.entity;

public class Transaction {
    public String date;
    public String transactionId;
    public String type;
    public double amount;
    public double balance;

    public Transaction(String date, String transactionId, String type, double amount, double balance) {
        this.date = date;
        this.transactionId = transactionId;
        this.type = type;
        this.amount = amount;
        this.balance = balance;
    }
}
